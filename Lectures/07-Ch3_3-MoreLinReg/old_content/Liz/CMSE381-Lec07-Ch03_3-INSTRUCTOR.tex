\begin{frame}[c]{Linear Regression with Multiple Variables}

\begin{columns}
\column{.45\textwidth}
\centering
\includegraphics[width = .5\textwidth]{Fig3-4}
\begin{itemize}
	\item Predict $Y$ on a multiple variables $X$
\begin{equation*}
	Y = \beta_0 + \beta_1 X_1 + \beta_2X_2 + \cdots \beta_p x_p + \e
\end{equation*}
	\item Find good guesses for $\hat \beta_0$, $\hat \beta_1,\cdots $.
\item $	\hat y_i = \hat \beta_0 + \hat \beta_1 x_i + \cdots + \hat \beta_px_p$
\end{itemize}
\column{.4\textwidth}
\centering
\begin{itemize}
\item $e_i = y_i - \hat y_i$ is the $i$th residual
\item RSS $= \sum_i e_i^2$
	\item RSS is minimized at \textit{least squares coefficient estimates}
\end{itemize}

\end{columns}

\end{frame}

\begin{frame}[c]{Assessing the accuracy of the module}{Almost the same as before}
	\centering
\begin{columns}
\column{.4\textwidth}
% \centering
\textbf{Residual standard error (RSE):}
\begin{align*}
	RSE &= \sqrt{ \frac{1}{n-{\color{red}p-1}} RSS}\\
\end{align*}
\InstructorNotes{measure lack of fit; }



\column{.45\textwidth}

\textbf{R squared:}
\begin{align*}
	R^2 &= \frac{TSS-RSS}{TSS} = 1-\frac{RSS}{TSS}
\end{align*}
% where total sum of squares is
\begin{equation*}
	TSS = \sum_i (y_i - \overline y)^2
\end{equation*}
\InstructorList{
\item Proportion in [0,1]; variability in $Y$ that can be explained using $X$
\item 0 means $Y$ not explained by linear regression on $X$}
\end{columns}

\end{frame}

\begin{frame}[c]{$R^2$ on Advertising data}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Just TV: $R^2 = 0.61$
	\item Just TV and radio: $R^2 = 0.89719$
	\item All three variables: $R^2 = 0.8972$
\end{itemize}

\column{.45\textwidth}
\centering
\InstructorList{
\item $R^2$ always increases with more variables, even if those variables are weakly associated to the response
\item Adding another variable decreases RSS on training
\item Small increase in $R^2$ if we include newspaper to other two, so can safely ignore
\item Large increase going from just TV to TV and Radio, so likely want to keep that
}
\end{columns}

\end{frame}

\begin{frame}[c]{Q4: Making predictions}

\begin{columns}
\column{.55\textwidth}
% \centering
Given estimates $\hat \beta_0,\cdots, \hat \beta_p$ for $\beta_0, \cdots, \beta_p$

Least squares plane:
\begin{equation*}
	\hat Y = \hat \beta_0 + \hat \beta_1 X_1 + \cdots + \hat \beta_p X_p
\end{equation*}
estimate for the true population regression plane
\begin{equation*}
	f(X) =  \beta_0 +  \beta_1 X_1 + \cdots +  \beta_p X_p
\end{equation*}

\centering

\begin{frame}[t]{Confidence vs Prediction Model}
	\begin{columns}
	\column{.45\textwidth}
	\centering
	\textbf{Confidence Interval}
	
	
	The range likely to contain the population parameter (mean, standard deviation) of interest.
	
	
	\qquad
	
	\textbf{Prediction Interval}
	
	The range that likely contains the value of the dependent variable for a single new observation given specific values of the independent variables.
	\column{.45\textwidth}
	\centering
	\includegraphics[width = \textwidth]{Advertising-TV-With-ConfidenceInterval}
	\end{columns}
	\end{frame}

\begin{frame}[t]{Specific to the Advertising Data}

	
	\begin{columns}
	\column{.4\textwidth}
	% % \centering
	\textbf{Confidence interval}: 
	quantify the uncertainty surrounding	the average sales over a large number of cities.
	
	
	
	\qquad
	
	\textbf{Advertising example:} \\
	If \$100K is spent on TV, and \$20K on radio, {\color{red}\textbf{in each of $n$ cities}} 
	
	\qquad 

	95\% CI for sales: 
	
	[10,985,\, 11,528].

	\tiny
	 \InstructorNotes{
	  Interpretation: if we collect a large number of data sets like the Advertising data
	 set, and we construct a confidence interval for the average sales on the basis of each
	 data set (given \$100,000 in TV and \$20,000 in radio advertising), then 95 \% of these 	 confidence intervals will contain the true value of average sales.
	 }
	
	
	\column{.4\textwidth}
	% % \centering
	\textbf{Prediction Interval:}
	quantify the uncertainty in sales for a particular city.
	
	\qquad
	
	\textbf{Advertising example:} \\
	Given that \$100,000 is spent on TV advertising and \$20,000 is spent on radio advertising in {\color{red}\textbf{Gotham City}}\\

	\qquad 

	% \centering
	95\% prediction interval for Gotham:  

	  [7,930, 14,580].
	
	\InstructorNotes{We interpret this to mean that 95\% of intervals of this form will contain the true value of Y for this city.}
	
	\end{columns}
	% \InstructorNotes{Both centered at $\hat f(X) = 11,256$}
	\end{frame}

\begin{frame}[t]{One-hot encoding}

\begin{columns}

Create a new variable
\begin{equation*}
		x_i =
	\begin{cases}
		1 & \text{if $i$th person is a student}\\
		0 & \text{if $i$th person is not a student}
	\end{cases}
\end{equation*}
% \column{.45\textwidth}
Model:
\begin{align*}
y_i &= \beta_0 + \beta_1x_i+\e_i\\
&= 
	\begin{cases}
		\beta_0 + \beta_1+\e_i	 & \text{if $i$th person is student}\\
		\beta_0 +\e_i & \text{if $i$th person isn't}
	\end{cases}
\end{align*}
\tiny
\InstructorList{
\item $\beta_0$ is avg credit card balance among those who are not a student 
\item $\beta_0 + \beta_1$ is the avg for those are
\item $\beta_1$ is avg diff in balance
}
\end{columns}
