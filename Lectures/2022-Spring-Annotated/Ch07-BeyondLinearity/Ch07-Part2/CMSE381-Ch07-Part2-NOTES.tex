\end{equation*}

		\item $b_3(x) = x^3$
		\item $\vdots$
		\item $b_d(x) = x^d$
	\end{itemize}

\footnotesize

		\item $b_1(x) = \mathrm{I}(X<c_1)$

	\end{itemize}

		\item $b_4(x) = (x-\xi_1)^3_+$
		\item $\vdots$
		\item $b_{K+3}(x) = (x-\xi_K)^3_+$


	\end{itemize}

\begin{frame}[t]{Cubic Spline}

\begin{columns}
\column{.65\textwidth}
\centering
Collection of piecewise cubics
\begin{equation*}
	f(x) =
	\begin{cases}
		\beta_{0,0} + \beta_{0,1} x + \beta_{0,2} x^2 + \beta_{0,3} x^3 & \text{if } x \leq \xi_1\\
		\beta_{1,0} + \beta_{1,1} x + \beta_{1,2} x^2 + \beta_{1,3} x^3 & \text{if } \xi_1 \leq x < \xi_2\\
		\beta_{2,0} + \beta_{2,1} x + \beta_{2,2} x^2 + \beta_{2,3} x^3 & \text{if }\xi_2 \leq x < \xi_3\\
		\beta_{3,0} + \beta_{3,1} x + \beta_{3,2} x^2 + \beta_{3,3} x^3 & \text{if } \xi_3 \leq x < \xi_4\\
		\hspace{1in}\vdots \\
		\beta_{K,0} + \beta_{K,1} x + \beta_{K,2} x^2 + \beta_{K,3} x^3 & \text{if } x \geq \xi_K\\
	\end{cases}
\end{equation*}

\column{.3\textwidth}
\centering
Require:
\begin{itemize}
	\item Continuous
	\item First derivative continuous
	\item Second derivative continuous
\end{itemize}

\section{Smoothing Splines}
%=====
\begin{frame}[t]{The goal}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item
Say we're looking for a function $g(x)$ that fits observed data well (i.e.~$RSS = \sum y_i - g(x_i))^2$ is small).
\item With no constraints on $g$.... \InstructorNotes{Just pick a function that interpolates all of the $y_i$}
\vspace{.5in}
\item So what we really want is.... \InstructorNotes{a function that is both smooth and has small RSS}
\end{itemize}
\column{.45\textwidth}
\centering

\end{columns}

\end{frame}
%--- Next Frame ---%


\begin{frame}[t]{Smoothing spline }

\begin{columns}
\column{.45\textwidth}
\centering
Find $g$ that mimizes
\begin{equation*}
	\sum_{i=1}^n (y_i-g(x_i))^2 + \lambda \int g''(t)^2 \,dt
\end{equation*}
\InstructorList{
\item $\lambda$ is the tuning parameter
\item The function $g$ that minimizes this is called the smoothing spline
\item This is another example of "Loss $+$ penalty" setting like ridge and lasso
\item $\lambda \int g''(t)^2 \,dt$ penalizes variability in $g$,
\item Roughness: $g''$ is large absolute value if $g(t)$ is wiggly near $t$
\item Integral is essentially averaging $g''$, so is small if overall not rough
}
\column{.45\textwidth}
\centering



\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{More on the smoothing spline }

\begin{columns}
\column{.45\textwidth}
\centering
Find $g$ that minimizes
\begin{equation*}
	\sum_{i=1}^n (y_i-g(x_i))^2 + \lambda \int g''(t)^2 \,dt
\end{equation*}

\column{.45\textwidth}
\centering

\InstructorList{
  \item $\lambda = 0$ means penalty has no effect
	\item $\lambda \to \infty$ means $g$ perfectly smooth (straight line)
	\item The function $g$ that minimizes can be shown to be piecewise cubic polynomial with knots at values $x_1,\cdots,x_n$ from the data, and with continuous first and second derivatives, and linear outside range
	\item AKA $g$ is a cubic spline with knots at all $x_i$
	\item Not quite the same as previous, since there is a $\lambda$ controlling the shrinkage
}

\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Effective degrees of freedom}

\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item Smoothing spline is natural cubic spline with knot at every $x_i$...... so too much flexibility?
	\item Actually $\lambda$ controls roughness and thus \textit{effective degrees of freedom} written $df_\lambda$
	\item $\lambda$ from 0 to $\infty$ makes effective deg freedom go from $n$ down to 2
	\item
}

\column{.45\textwidth}
\centering

\InstructorNotes{Why effective instead of just DF?}
\InstructorList{
  \item Smoothing splines have $n$ parameters so $n$ degrees of freedom
	\item However, with $\lambda$, these are heavily shrunk down/constrained

}

\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Computing $df_\lambda$}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Fix $\lambda$
	\item Find $g$ that minimizes
	\begin{equation*}
		\sum_{i=1}^n (y_i-g(x_i))^2 + \lambda \int g''(t)^2 \,dt
	\end{equation*}
	\item Write $\hat {\boldsymbol g}_\lambda$ for the $n$-vector with fitted values of smoothing spline at training points $x_1,\cdots,x_n$
\end{itemize}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Find \InstructorNotes{(there is an ugly formula)} $n \times n$ matrix ${\boldsymbol S}_\lambda$ where
	\begin{equation*}
		\hat {\boldsymbol g}_\lambda =  {\boldsymbol S}_\lambda \boldsymbol{y}
	\end{equation*}
	\item The effective degrees of freedom is defined to be
	\begin{equation*}
		df_\lambda = \sum_{i=1}^n \{ {\boldsymbol S}_\lambda\} _{ii} = \mathrm{trace}({\boldsymbol S}_\lambda)
	\end{equation*}
\end{itemize}
\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Choosing $\lambda$}
	{When in doubt, Cross-Validate}

\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item We don't have to pick the knots (yay!)
  \item We want LOOCV
	\item Cheap trick for that lets us get the LOOCV error with same cost as computing for a single fit
	\item similar in spirit to what we did with linear regression in Ch 5
}


\column{.45\textwidth}
\centering

\InstructorList{
\item Middle is LOOCV normal version
  \item $\hat g _{\lambda}^{-i}(x_i)$ indicates the fitted value for this smoothing spline evaluated at $x_i$, where the fit uses all of the training observations except
for the ith observation $(x _i , y _i )$.
\item Right side is the trick
\item $\hat g_\lambda (x_i) $ indicates the smoothing
spline function fit to all of the training observations and evaluated at $x_i$ .
}

\end{columns}

\end{frame}

\section{General Additive Models (GAMs)}
%=====
\begin{frame}[c]{Yet another way to extend linear models}

\begin{columns}
\column{.45\textwidth}
\centering
Learning $Y$ using $X_1,\cdots, X_p$

\vspace{12pt}

\textbf{Old version:}
\begin{equation*}
	y_i = \beta_0 + \beta_1 x_{i1} + \beta_2 x_{i2} + \cdots + \beta_p x_{ip}+\e_i
\end{equation*}
\InstructorNotes{Replace each $\beta_j x_{ij}$ by $f_j(x_i)$}

\vspace{12pt}

\textbf{GAM:}
\begin{align*}
	y_i &= \beta_0 + \sum_{j=1}^p f_j(x_{ij})+\e_i\\
	 &= \beta_0 + f_1(x_{i1}) + f_2(x_{i2}) + \cdots + f_d(x_{id}) + \e_i
\end{align*}
\column{.45\textwidth}
\centering

\InstructorList{
  \item This is an example of a GAM
	\item it is called additive because we get a diff $f_j$ for each $X_j$
	\item
}

\end{columns}

\end{frame}

\begin{frame}[t]{GAMs Pros}

\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
  \item Note that we can use more than splines and dummy variables...... basically anything can be used as input
	\item Allow non-linear $f_j$ for  each $X_j$.
	No need to manually try out many different transformations on each variable individually.
\item The non-linear fits can potentially make more accurate predictions for the response $Y$.

}

\column{.45\textwidth}
\centering

\InstructorList{
\item Additive means easy interpretation: effect of each $X_j$
on $Y$ individually while holding all of the other variables fixed.
\item The smoothness of the function $f_j$ for the variable $X_j$ can be summarized via degrees of freedom.
}


\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[t]{GAMs Cons/Limitations}

\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
  \item Restricted to be additive
	\item With many variables, important interactions can be missed.

}
\column{.45\textwidth}
\centering

\InstructorList{
\item We can manually add interaction terms to the GAM model by including additional predictors of the
form $X_j \cdot X_k$.
\item We can add low-dimensional interaction functions of the form $f_{jk} (X_j , X_k )$ into the model; such terms can be fit using two-dimensional smoothers such as local regression, or two-dimensional splines (not covered here)
}

\end{columns}

\end{frame}
