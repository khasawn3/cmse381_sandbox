\begin{equation*}
	y_i = \beta_0 + \beta_1b_1(x_i) + \beta_2 b_2(x_i) + \cdots + \beta_K b_K(x_i) + \e_i
\end{equation*}

\begin{columns}
\column{.25\textwidth}
\centering
Polynomial Regression:
\footnotesize
	\begin{itemize}
		\item $b_1(x) = x$
		\item $b_2(x) = x^2$
		\item $b_3(x) = x^3$
		\item $\vdots$
		\item $b_d(x) = x^d$
	\end{itemize}


\column{.35\textwidth}
\centering
Step functions
\footnotesize
	\begin{itemize}
		\item $b_1(x) = \mathrm{I}(X<c_1)$
		\item $b_2(x) = \mathrm{I}(c_1 \leq X < c_2)$
		\item $b_2(x) = \mathrm{I}(c_2 \leq X < c_3)$
		\item $\vdots $
		\item $b_{K-1}(x) = \mathrm{I}(c_{K-1} \leq X < c_K)$
		\item $b_K(x) = \mathrm{I}(c_K \leq X)$

	\end{itemize}



\column{.35\textwidth}
\centering
Cubic splines

		\item $b_4(x) = (x-\xi_1)^3_+$
		\item $\vdots$
		\item $b_{K+3}(x) = (x-\xi_K)^3_+$


	\end{itemize}

\begin{frame}[t]{Cubic Spline}

\begin{columns}
\column{.65\textwidth}
\centering
Collection of piecewise cubics
\begin{equation*}
	f(x) =
	\begin{cases}
		\beta_{0,0} + \beta_{0,1} x + \beta_{0,2} x^2 + \beta_{0,3} x^3 & \text{if } x \leq \xi_1\\
		\beta_{1,0} + \beta_{1,1} x + \beta_{1,2} x^2 + \beta_{1,3} x^3 & \text{if } \xi_1 \leq x < \xi_2\\
		\beta_{2,0} + \beta_{2,1} x + \beta_{2,2} x^2 + \beta_{2,3} x^3 & \text{if }\xi_2 \leq x < \xi_3\\
		\beta_{3,0} + \beta_{3,1} x + \beta_{3,2} x^2 + \beta_{3,3} x^3 & \text{if } \xi_3 \leq x < \xi_4\\
		\hspace{1in}\vdots \\
		\beta_{K,0} + \beta_{K,1} x + \beta_{K,2} x^2 + \beta_{K,3} x^3 & \text{if } x \geq \xi_K\\
	\end{cases}
\end{equation*}

\column{.3\textwidth}
\centering
Require:
\begin{itemize}
	\item Continuous
	\item First derivative continuous
	\item Second derivative continuous
\end{itemize}

\section{Smoothing Splines}
%=====
\begin{frame}[t]{The goal}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item
Say we're looking for a function $g(x)$ that fits observed data well (i.e.~$RSS = \sum y_i - g(x_i))^2$ is small).
\item With no constraints on $g$.... \InstructorNotes{Just pick a function that interpolates all of the $y_i$}
\vspace{.5in}
\item So what we really want is.... \InstructorNotes{a function that is both smooth and has small RSS}
\end{itemize}
\column{.45\textwidth}
\centering

\end{columns}

\end{frame}
%--- Next Frame ---%


\begin{frame}[t]{Smoothing spline }

\begin{columns}
\column{.45\textwidth}
\centering
Find $g$ that mimizes
\begin{equation*}
	\sum_{i=1}^n (y_i-g(x_i))^2 + \lambda \int g''(t)^2 \,dt
\end{equation*}
\InstructorList{
\item $\lambda$ is the tuning parameter
\item The function $g$ that minimizes this is called the smoothing spline
\item This is another example of "Loss $+$ penalty" setting like ridge and lasso
\item $\lambda \int g''(t)^2 \,dt$ penalizes variability in $g$,
\item Roughness: $g''$ is large absolute value if $g(t)$ is wiggly near $t$
\item Integral is essentially averaging $g''$, so is small if overall not rough
}
\column{.45\textwidth}
\centering



\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{More on the smoothing spline }

\begin{columns}
\column{.45\textwidth}
\centering
Find $g$ that minimizes
\begin{equation*}
	\sum_{i=1}^n (y_i-g(x_i))^2 + \lambda \int g''(t)^2 \,dt
\end{equation*}

\column{.45\textwidth}
\centering

\InstructorList{
  \item $\lambda = 0$ means penalty has no effect
	\item $\lambda \to \infty$ means $g$ perfectly smooth (straight line)
	\item The function $g$ that minimizes can be shown to be piecewise cubic polynomial with knots at values $x_1,\cdots,x_n$ from the data, and with continuous first and second derivatives, and linear outside range
	\item AKA $g$ is a cubic spline with knots at all $x_i$
	\item Not quite the same as previous, since there is a $\lambda$ controlling the shrinkage
}

\end{columns}

\end{frame}

\begin{frame}[c]{Choices to be made for local regression}

\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
  \item Weighting function $K$
	\item Whether linear/constant/quadratic in Step 3 (linear as written)
	\item Most important: span $s$, proportion of points used to compute local regression at $x_0$
	\item $s$ controls flexibility of the non-linear fit
	\item Small $s$: local and wiggly fit
	\item Big $s$ global fit of all training data
}
