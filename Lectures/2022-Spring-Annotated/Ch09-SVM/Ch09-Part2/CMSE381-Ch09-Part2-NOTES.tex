\begin{frame}[t]{How do we pick? Maximal margin classifier}

\begin{columns}
\column{.45\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Fig9-3}
\column{.45\textwidth}
\centering
\InstructorList{
\item \textit{Maximal margin hyperplane} is the hyperplane with the largest minimum distance to the training observations
\item Classify using it gives you the \textit{maximal margin classifier}
\item Observations that are closest are called \textit{support vectors}
\item If points moved slightly then MMH would move
\item No effect if any other points moved
}

\end{columns}

\end{frame}

\InstructorNotes{There's lots of separating hyperplanes here, so how do we find a "best" one?}

\item Smallest such distance is the \textit{margin}
}

  \item Last eq forces points to be on the correct side of the hyperplane if $M>0$

\begin{frame}[t]{Inner products}
\begin{equation*}
	\langle a,b \rangle = \sum_{i=1}^r a_ib_i
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item Inner product of two observations $x_i,x_{i'}$ is
	\begin{equation*}
		\langle x_i,x_{i'} \rangle = \sum_{j=1}^p x_{ij}x_{i'j}
	\end{equation*}
	\item Given a point $p$ and the unit normal vector to the line $v$, the distance is $\langle p, v \rangle + \beta_0$ IS THIS TRUE?

}

\column{.45\textwidth}
\centering

\InstructorList{
\item For an example, you can use $Y + X -2 = 0$, dividing by $\sqrt 2$ to make unit vector
		\item Then check $-2/\sqrt{2} + \langle (1/\sqrt{2},1/\sqrt{2}),(2,1)\rangle = 2/\sqrt{2}$ to get distance to the point $(2,1)$
}

\end{columns}

\end{frame}

\begin{frame}[t]{SVC via inner products}

\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item Writing what we had before, $f(x) = \beta_0 + \beta_1x_1 + \cdots + \beta_px_p = \beta_0 + \langle \beta, x\rangle$
	\item Via some magic, there are coefficients for which
	\begin{equation*}
		f(x) = \beta_0 + \sum_{i=1}^n \alpha_i \langle x, x_i \rangle
	\end{equation*}
	meaning one for every observation
	\item To estimate the paramters $\alpha_1,\cdots,\alpha_n$ and $\beta_0$, need $\binom{n}{2} = n(n-1)/2$ inner products $\langle x_i,x_{i'} \rangle$
}

\column{.45\textwidth}
\centering

\end{columns}

\end{frame}

\begin{frame}[t]{SVC via inner products of support vectors}

\begin{columns}
\column{.45\textwidth}
\centering

\begin{equation*}
	f(x) = \beta_0 + \sum_{i \in \mathcal{S}} \alpha_i\langle x, x_i \rangle
\end{equation*}
\InstructorList{
  \item Turns out $\alpha_i$ are only nonzero for support vectors
	\item Uses fewer terms, so way fewer computations to figure out $f(x)$ for some new $x$
}

\begin{frame}[t]{The kernel}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{equation*}
	K(x_i,x_i')
\end{equation*}
\InstructorList{
  \item Swap out my inner product $\langle x, x_i \rangle$ for something potentially more complicated
	\item $\langle x, x_i \rangle$ is known as a linear kernel
	\item near kernel essentially quantifies the similarity of a
pair of observations using Pearson (standard) correlation
}

\column{.45\textwidth}
\centering
\begin{equation*}
	f(x) = \beta_0 + \sum_{i \in \mathcal{S}}\alpha_i K(x,x_i)
\end{equation*}

\InstructorList{
  \item The function defined is as above with whatever choice of $K$
	\item When the support vector classifier is combined with a non-linear kernel the resulting classifier is 	known as a support vector machine.
}

\end{columns}

\end{frame}

\begin{frame}[t]{A polynomial kernel}
\begin{equation*}
	K(x_i,x_{i'}) = \left(1 + \sum_{j=1}^p x_{ij}x_{i'j} \right)^d
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item much more flexible decision boundary.
	\item amounts to fitting a support vector
classifier in a higher-dimensional space involving polynomials of degree $ d$,
rather than in the original feature space.
\item Right: degree 3 polynomial kernel

}

\column{.35\textwidth}
\centering
\includegraphics[width = \textwidth]{Fig9-9a}
\end{columns}

\end{frame}

\begin{frame}[t]{A radial kernel}
\begin{equation*}
	K(x_i,x_i') = \exp \left( -\gamma \sum_{j=1}^p(x_{ij} - x_{i'j})^2 \right)
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering

\InstructorList{
  \item Draw picture of circular values around a point
	\item big distance makes $\sum (x_{ij}-x_{i'j})^2$ big, but then $e^{-big}$ is tiny.
	\item In $f(x)$, this means that $x_i$ plays little role if it's far away.
	\item Radial kernel has local behavior
}

\column{.35\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Fig9-9b}
\end{columns}

\end{frame}
