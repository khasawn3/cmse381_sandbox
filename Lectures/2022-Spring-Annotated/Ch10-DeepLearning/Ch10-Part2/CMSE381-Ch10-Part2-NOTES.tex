\begin{frame}[t]{Choices for activation function }

\begin{columns}
\column{.45\textwidth}
\centering
Sigmoid:
\begin{equation*}
	g(z) = \frac{e^z}{1+e^z} = \frac{1}{1+e^{-z}}
\end{equation*}
\includegraphics[width = \textwidth]{DL-sigmoid}
\InstructorList{
  \item Converts linear function in probabilities in [0,1]
}

\column{.45\textwidth}
\centering
ReLU: Rectified linear unit
\begin{equation*}
	g(z) = (z)_+ =
	\begin{cases}
		0 & \text{if }z<0 \\
		z & \text{else.}
	\end{cases}
\end{equation*}
\includegraphics[width = \textwidth]{DL-ReLU}
\end{columns}

\end{frame}

\begin{frame}[t]{More notes on convolution}

\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
  \item Filters idea from image processing
	\item Novelty in DL is that the choice of filters are learned
	\item Filter weights: one hidden unit per pixel in convolved image; but more constrained than that
}

\column{.45\textwidth}
\centering
\InstructorList{
  \item Image is in color, so 3 channels
	\item This means the convolution filter will have three channels, one per color, each of dimension 3x3
	\item Results are summed to form a 2-dimensional output. Color is "forgotton" at this point
	\item $K$ different filters here means $K$ 2-d output maps, which can then be treated as channels
	\item Applying ReLU to convolved image. Sometimes viewed as separate layer in CNN, and then called a detector layer.
}

\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Pooling layers}

\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
  \item Condense a large image into a smaller summary image
	\item Lots of options, but max pooling is common:
	\item For each 2x2 non-overlapping block of pixels, get a single new pixel with the maximum value from the block
	\item Provides location invariance: so long as large value in one of the four pixels, the whole block registers as a large value
}

\column{.45\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Eq10-3-2.png}
\end{columns}

\end{frame}

\begin{frame}[t]{Putting it together to make a CNN}

\includegraphics[width = \textwidth, align = c]{Fig10-8}
\begin{columns}
\column{.45\textwidth}
\centering
\footnotesize
\InstructorList{
  \item Start with three channels
	\item get a new channel at first hidden layer for each filter. Padding to keep the same size
	\item Next is max pool layer,
	\item convolve \& pool several more times
	\item Since size of image decreased, we increase number of filters in the next convolve layer to compensate
	% \item Can repeat several convolves before pool, this effectively increases the dimension fo the feature
}

\column{.45\textwidth}
\centering
\footnotesize
\InstructorList{
  \item After this, the maps are flattened (treated as separate units )
	\item Fed into 1 or more fully-connected layers
	\item Last is softmax activation for the 100 classes
}

\end{columns}

\end{frame}
