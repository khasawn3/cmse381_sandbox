\begin{frame}[c]{Logistic Regression }

$$
\Pr(\texttt{default} = \texttt{yes} \mid \texttt{balance})
=
	\frac{e^{\beta_0 + \beta_1 \texttt{balance}}}{1 +e^{\beta_0 + \beta_1\texttt{balance}} }
$$

\begin{frame}[t]{Odds}
	\begin{equation*}
		\frac{p(x)}{1-p(x)}
		= \frac{\Pr(Y = 1 \mid X=x)}{1-\Pr(Y = 1 \mid X=x)}
		= \frac{\Pr(Y = 1 \mid X=x)}{\Pr(Y = 0 \mid X=x)}
	\end{equation*}
\begin{columns}
\column{.4\textwidth}
\centering
\InstructorList{
\item Logistic function is chosen so that odds are linear
\item Can take any value from 0 (low odds) to $\infty$ (high odds)
}
\column{.4\textwidth}
\centering
\InstructorList{
\item 1 in 4 people with odds of 1/3 will default since $p(X) = 0.25$ and $0.25/(1-0.25) = 1/3$
\item 9 in 10 people with odds of 9 will default since $p(X) = 0.9$ and $0.9/(1-0.9) = 9$
}
\end{columns}

\InstructorNotes{Draw this:
\includegraphics[width = .3\textwidth,align = c]{1024px-Probability_vs_odds.png}
}
\end{frame}

\begin{frame}[c]{How to get logistic function}

\begin{columns}
\column{.4\textwidth}
\centering
Assume the (natural) log odds (logits) follow a linear model

\begin{equation*}
	\log \left( \frac{p(x)}{1-p(x)} \right) = \beta_0 + \beta_1x
\end{equation*}


\column{.4\textwidth}
\centering
Solve for $p(x)$:
\begin{equation*}
	p(x) = \frac{e^{\beta_0 + \beta_1x}}{1+e^{\beta_0 + \beta_1x}}
\end{equation*}
\end{columns}

\bvfill

Playing with the logistic function: \url{https://www.desmos.com/calculator/jzsakksqcm}
\end{frame}

\begin{frame}[c]{Interpreting the coefficients}

\begin{columns}
\column{.4\textwidth}
\centering
\begin{equation*}
	p(x) = \frac{e^{\beta_0 + \beta_1x}}{1+e^{\beta_0 + \beta_1x}}
\end{equation*}

\begin{equation*}
	\log \left( \frac{p(x)}{1-p(x)} \right) = \beta_0 + \beta_1x
\end{equation*}
\column{.4\textwidth}
\centering
\InstructorNotes{$\beta_1$ means increasing $x$ by one unit increases the log odds by 1 unit}
\end{columns}

\begin{frame}[c]{Estimating Coefficients: Maximum Likelihood Estimation}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item
\textbf{Likelihood:} Probability that data is generated from a model
\begin{equation*}
	\ell(model) = \P[data \mid model]
\end{equation*}
\item Find the most likely model
\begin{equation*}
	\max_{model} l(model)
\end{equation*}
\item Hard to maximize likelihood, instead maximize log
\begin{equation*}
	\max_{model} \log (\ell(model))
\end{equation*}
\item Strictly increasing log function doesn't change maximum
\end{itemize}
\column{.45\textwidth}
\centering

\begin{align*}
	\Pr(Y=1 \mid X) &= p(X)\\
	&  =
	\frac{e^{\beta_0 + \beta_1X}}{1 +e^{\beta_0 + \beta_1X} }
\end{align*}
\begin{equation*}
	\ell(\beta_0,\beta_1) =
	\prod_{i \mid y_i = 1} p(x_i)
	\prod_{i'\mid y_{i'} = 0} (1-p(x_{i'}))
\end{equation*}

\InstructorList{
\item Last time I said no $\beta_i$'s..... they're hidden in the $p(x_i)$ function
% \item
}
\end{columns}

\end{frame}

\begin{frame}[c]{Bayes Theorem}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{equation*}
P(A\mid B)={\frac {P(A) \cdot P(B\mid A)}{P(B)}}
\end{equation*}
\column{.45\textwidth}
\centering
\InstructorList{
\item $P(A \mid B)$ Posterior, probability of A being true given B
\item $P(A)$ Prior, probability of $A$ being true
\item $P(B)$ Marginalization, probability of $B$ being true
\item $P(B \mid A)$ Likelihood, probability of $B$ true given that $A$ is true
}
\end{columns}

\end{frame}

\begin{frame}[c]{Following book notation}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Classify an observation into one of $K\geq 2 $ classes
	\item $\pi_k$ is the \textit{prior} probability that a randomly chosen observation comes from the $k$th class, $P(Y = k)$
\end{itemize}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item $f_k(X) = \Pr(X \mid Y = k)$ is the density function of $X$ for an observation from the $k$th class

	\InstructorList{
	\item Large $f_k(x)$ if there is high probability that observation in the $k$th class has $X \approx x$
	\item Small if unlikely that an observation in the $k$th class has $X \approx x$
	}
	\item $\Pr(X) = \sum_\ell \pi_\ell f_\ell(X)$
\end{itemize}

\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Bayes to the rescue!}

\begin{columns}
\column{.4\textwidth}
% \centering
Posterior probability that an observation $X=x$ belongs to the $k$th class:
\begin{equation*}
	p_k(x) = \Pr(Y = k \mid X = x ) = \frac{ \pi_k f_k(x)}{\sum_{\ell = 1}^K \pi_\ell f_\ell(x)}
\end{equation*}

\column{.4\textwidth}
\centering
\InstructorList{
\item Second equation is Bayes, $p_k(x)$ is our notation for it
\item Plug in estimates for $\pi_k$ and $f_k(x)$ to get an estimate
\item Estimate for $\pi_k$ is easy with a random sample, just take proportion that are $k$th class
\item Estimating desnity $f_k$ is hard(er).....
}

\end{columns}

\end{frame}

\section{Linear Discriminant Analysis for $p=1$}
%=============
\begin{frame}[c]{Assumptions}

\begin{columns}
\column{.45\textwidth}
\centering
Assume $f_k(x)$ is normal/Gaussian:

\begin{equation*}
	f_k(x) = \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma_k^2}(x-\mu_k)^2 \right)
\end{equation*}

\begin{itemize}
	\item $\mu_k$ = mean of $k$th class
	\item $\sigma^2_k$ = variance of $k$th class
	\item Assume $\sigma_1^2 = \cdots = \sigma_K^2$
\end{itemize}
\column{.45\textwidth}
\centering
\begin{equation*}
	p_k(x) = \frac{ \pi_k \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_k)^2 \right)}{\sum_{\ell = 1}^K \pi_\ell \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_\ell)^2 \right)}
\end{equation*}


\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Bayes Classifier}
	{Same Bayes person, different Bayes definition}

\begin{columns}
\column{.45\textwidth}
\centering

\textbf{Bayes classifier}

Assign the class $k$ for which $p_k(x)$ is largest



\column{.45\textwidth}
\centering

Finding largest $k$
\begin{equation*}
	p_k(x) = \frac{ \pi_k \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_k)^2 \right)}{\sum_{\ell = 1}^K \pi_\ell \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_\ell)^2 \right)}
\end{equation*}

[insert icky algebra]

is the same as finding largest $k$ for
\begin{equation*}
	\delta_k(x) = x \cdot \frac{\mu_k}{\sigma^2} - \frac{\mu_k^2}{2\sigma^2} + \log (\pi_k)
\end{equation*}
\end{columns}

\end{frame}

\begin{frame}[c]{New plan: Linear Discriminant Analysis (LDA)}

\begin{columns}
\column{.45\textwidth}
% \centering
Estimate
\begin{equation*}
	\delta_k(x) = x \cdot \frac{\mu_k}{\sigma^2} - \frac{\mu_k^2}{2\sigma^2} + \log (\pi_k)
\end{equation*}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item $\displaystyle{
	\hat \mu_k = \frac{1}{n_k} \sum_{i \mid y_i = k} x_i
	}$
	\item $\displaystyle{
	\hat \sigma^2 = \frac{1}{n-K}\sum_{k=1}^K \sum_{i \mid y_i = k}(x_i - \hat \mu_k)^2
	}$
	\item $\hat \pi_k = n_k/n$
\end{itemize}
\end{columns}
