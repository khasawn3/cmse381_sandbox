\section{LDA $p=1$ Review}
%=================


\begin{frame}[c]{Goal:}

\begin{columns}
\column{.45\textwidth}
\centering
% Approximate $Pr(Y = k \mid X = x)$

\begin{itemize}
	\item Classify an observation into one of $K\geq 2 $ classes
	\item $\pi_k$ is the \textit{prior} probability that a randomly chosen observation comes from the $k$th class, $P(Y = k)$
	\item $f_k(X) = \Pr(X \mid Y = k)$ is the density function of $X$ for an observation from the $k$th class

	\item $\Pr(X) = \sum_\ell \pi_\ell f_\ell(X)$
\end{itemize}

\column{.45\textwidth}
\centering


\begin{block}
	{Bayes Theorem}
\begin{equation*}
P(A\mid B)={\frac {P(A) \cdot P(B\mid A)}{P(B)}}
\end{equation*}
\end{block}
Posterior probability that an observation $X=x$ belongs to the $k$th class:
\begin{equation*}
	p_k(x) = \Pr(Y = k \mid X = x ) = \frac{ \pi_k f_k(x)}{\sum_{\ell = 1}^K \pi_\ell f_\ell(x)}
\end{equation*}
\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Linear Discriminant Analysis for $p=1$}

\begin{columns}
\column{.45\textwidth}
Assume $f_k(x)$ is normal/Gaussian:

\begin{equation*}
	f_k(x) = \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma_k^2}(x-\mu_k)^2 \right)
\end{equation*}

\begin{itemize}
	\item $\mu_k$ = mean of $k$th class
	\item $\sigma^2_k$ = variance of $k$th class
	\item Assume $\sigma_1^2 = \cdots = \sigma_K^2$
\end{itemize}
\begin{equation*}
	p_k(x) = \frac{ \pi_k \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_k)^2 \right)}{\sum_{\ell = 1}^K \pi_\ell \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_\ell)^2 \right)}
\end{equation*}
\column{.45\textwidth}
\centering


Finding  $k$ with largest $p_k(x)$ is the same as finding $k$ for which largest
\begin{equation*}
	\delta_k(x) = x \cdot \frac{\mu_k}{\sigma^2} - \frac{\mu_k^2}{2\sigma^2} + \log (\pi_k)
\end{equation*}

\begin{itemize}
	\item $\displaystyle{
	\hat \mu_k = \frac{1}{n_k} \sum_{i \mid y_i = k} x_i
	}$
	\item $\displaystyle{
	\hat \sigma^2 = \frac{1}{n-K}\sum_{k=1}^K \sum_{i \mid y_i = k}(x_i - \hat \mu_k)^2
	}$
	\item $\hat \pi_k = n_k/n$
\end{itemize}
\end{columns}

\end{frame}

\section{LDA for $p \geq 1$}
%=================

\begin{frame}[c]{Multivariate Gaussians}

Assume $X = (X_1,X_2,\cdots,X_p)$ is drawn from a multivariate Gaussian. $X \sim N(\mu,\Sigma)$

\centering
\includegraphics[width = .8\textwidth, align = c]{Fig4-5}
\begin{columns}
\column{.45\textwidth}
\centering
$\Var(X_1) = \Var(X_2)$

$\mathrm{Cor}(X_1,X_2) = 0$
\column{.45\textwidth}
\centering
$\Var(X_1) \neq \Var(X_2)$

$\mathrm{Cor}(X_1,X_2) \neq  0$

\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{More on Multivariate Gaussians}

% Assume $X = (X_1,X_2,\cdots,X_p)$ is drawn from a multivariate Gaussian.
\begin{equation*}
	f(x) = \frac{1}{(2\pi)^{p/2} |\Sigma|^{1/2}} \exp \left( -\frac{1}{2} (x-\mu)^T \Sigma^{-1}(x-\mu) \right)
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item $X \sim N(\mu,\Sigma)$
	\item $E(X) = \mu $ vector of means
	\item $\mathrm{Cov}(X) = {\Sigma} = $ $p \times p$ Covariance matrix
\end{itemize}
\column{.45\textwidth}
\centering
\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Assumptions for LDA}

\begin{columns}
\column{.45\textwidth}
\centering

\begin{itemize}
	\item $p>1$ predictors $X_1,\cdots,X_p$
	\item Assume observations in the $k$th class $\sim N(\mu_k,\Sigma)$
	\InstructorList{
	\item $\mu_k$ is class specific mean
	\item $\Sigma$ is common to all $K$ classes
	}
\end{itemize}
\column{.45\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Fig4-6a}
\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Bayes decision boundary}

\begin{columns}
\column{.45\textwidth}
\centering

Combine density for $k$th class $f_k(X=x)$ with
\begin{equation*}
	p_k(x) = \Pr(Y = k \mid X = x ) = \frac{ \pi_k f_k(x)}{\sum_{\ell = 1}^K \pi_\ell f_\ell(x)}
	% p_k(x) = \frac{ \pi_k \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_k)^2 \right)}{\sum_{\ell = 1}^K \pi_\ell \frac{1}{\sqrt{2 \pi}}\exp \left(\frac{-1}{2\sigma^2}(x-\mu_\ell)^2 \right)}
\end{equation*}
to get

\begin{block}
	{Bayes classifier}

	Return the class $k$ for which
\begin{equation*}
	\delta_k(x) =
	x^T \Sigma \inv \mu_k  - \frac{1}{2} \mu_k^T \Sigma\inv \mu_k + \log \pi_k
\end{equation*}
is largest.
\end{block}
\InstructorNotes{Note that $\delta_k(x)$ is actually linear function, hence the L in LDA}

\column{.45\textwidth}
\centering


\includegraphics[width = \textwidth, align = c]{Fig4-6a}


\end{columns}

\end{frame}

\section{Naive Bayes}
%=======
\begin{frame}[t]{Naive Bayes Assumptions}


	\begin{equation*}
		p_k(x) = \Pr(Y = k \mid X = x ) = \frac{ \pi_k f_k(x)}{\sum_{\ell = 1}^K \pi_\ell f_\ell(x)}
	\end{equation*}
\begin{columns}
\column{.4\textwidth}
\centering
Assume the features are independent:
\begin{align*}
	f_k(x) &= f_{k1}(x_1) \times f_{k2}(x_2) \times \cdots \times f_{kp}(x_p) \\
	&= \prod_{j=1}^p f_{jk}(x_j)
\end{align*}
\column{.4\textwidth}
\centering
\InstructorList{
\item Note that no longer have a Gaussian assumption
\item No need to worry about figuring out association between the p predictors, simplifies estimation
\item Reasonable assumption? Probably not. But if $n$ is NOT large enough relative to $p$ to estimate joint distribution of predictors within class, this is a reasonable alternative.
\item Naive Bayes introduces some bias, but reduces variance, so works well in practice

}
\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Gaussian Naive Bayes}

\begin{columns}
\column{.4\textwidth}
\centering
Let's assume:
\begin{itemize}
	\item Features are independent
	\item They come from a Gaussian, $X_j \mid Y = k \sim N(\mu_{jk},\sigma^2_{jk})$
\end{itemize}
\column{.4\textwidth}
\centering
\InstructorList{
\item Looks like QDA, but has the independence assumption
\item Result is class-specific  covariance matrix is diagonal, just has variance along the diagonal
}
\end{columns}

\end{frame}
