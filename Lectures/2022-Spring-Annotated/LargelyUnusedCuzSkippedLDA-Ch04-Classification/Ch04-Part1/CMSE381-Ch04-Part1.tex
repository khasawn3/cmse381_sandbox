\section{Classification Overview}
%=================

\begin{frame}[t]{What is classification }

\begin{columns}
\column{.45\textwidth}
% \centering
Classification: When the response variable is qualitative

\InstructorList{
\item qualitative: Unordered set
\item Examples
\InstructorList{
\item Eye color $\in \{ brown, blue, green\}$
\item $email \in \{ spam, not spam\}$
}
}
\column{.45\textwidth}
% \centering
\begin{itemize}
	\item
Given feature vector $X$ and qualitative response $Y$ in the set $S$, the goal is to find a function (classifier) $C(X)$ taking $X$ as input and predicting its value for $Y$.

	\item
We are more interested in estimating the probabilities that X belongs to each category
\end{itemize}
\end{columns}

\end{frame}

\begin{frame}[t]{Let's just use regression!}
	{JK that's a bad idea}

\begin{columns}
\column{.45\textwidth}
\centering
\textbf{Bad idea:}
\begin{itemize}
	\item Set $Y$ to be a dummy variable taking values in $\{0,1,2,\cdots\}$
	\item Run regression, and choose $k$ based on what integer value $\hat y$ is closest to
\end{itemize}
\column{.45\textwidth}
% \centering

Ex.
\begin{equation*}
	Y =
	\begin{cases}
		1 & \text{if stroke }\\
		2 & \text{if drug overdose }\\
		3 & \text{if epileptic seizure}
	\end{cases}
\end{equation*}
vs.
\begin{equation*}
	Y =
	\begin{cases}
		1 & \text{if mild }\\
		2 & \text{if moderate }\\
		3 & \text{if severe}
	\end{cases}
\end{equation*}
\end{columns}
\InstructorNotes{Ordering implies something about closeness. Draw a number line to emphasize this distinction. }
\end{frame}

\begin{frame}[c]{Bad idea is still not a great idea for two levels}

\begin{columns}
\column{.45\textwidth}
\centering

\begin{equation*}
	Y =
	\begin{cases}
		0 & \text{if stroke}\\
		1 & \text{if overdose}\\
	\end{cases}
\end{equation*}
\begin{itemize}
	\item Fit linear regression
	\item Predict overdose if $\hat y > 0.5$; stroke otherwise

\begin{frame}[c]{Logistic Regression }


\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Let $p(X) = \Pr(Y =1 \mid X)$.
	\item Turn this into something with range $\R$
	\begin{equation*}
		\ln \left( \frac{p(X)}{1-p(X)} \right)  = \beta_0 + \beta_1X
	\end{equation*}
	\item Called \textit{log odds}, or \textit{logit}
	\item Solving for $p(X)$ gets
\begin{equation*}
	p(X) =
	\frac{e^{\beta_0 + \beta_1X}}{1 +e^{\beta_0 + \beta_1X} }
\end{equation*}
\end{itemize}
\column{.45\textwidth}
\centering

\end{columns}

\end{frame}

\begin{frame}[c]{Estimating Coefficients: Maximum Likelihood Estimation}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item
\textbf{Likelihood:} Probability that data is generated from a model
\begin{equation*}
	\ell(model) = \P[data \mid model]
\end{equation*}
\item Find the most likely model
\begin{equation*}
	\max_{model} l(model)
\end{equation*}
\item Hard to maximize likelihood, instead maximize log
\begin{equation*}
	\max_{model} \log (\ell(model))
\end{equation*}
\item Strictly increasing log function doesn't change maximum
\end{itemize}
\column{.45\textwidth}
\centering

\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[t]{Maximum Likelihood for Logistic Regression}

\begin{columns}
\column{.45\textwidth}
\centering

\begin{align*}
	\Pr(Y=1 \mid X) &= p(X)\\
	&  =
	\frac{e^{\beta_0 + \beta_1X}}{1 +e^{\beta_0 + \beta_1X} }
\end{align*}
\column{.45\textwidth}
\centering
\begin{equation*}
	\ell(\beta_0,\beta_1) =
	\prod_{i \mid y_i = 1} p(x_i)
	\prod_{i'\mid y_{i'} = 0} (1-p(x_{i'}))
\end{equation*}
\end{columns}

\InstructorNotes{
Choose $\hat \beta_0$ and $\hat \beta_1$ to maximize this function. Math outside the scope of class, computer magic finds it for us
}

\end{frame}

\begin{frame}[c]{Multiple Logistic Regression}

\begin{columns}
\column{.65\textwidth}
\centering
\textbf{Multiple features:}
\begin{equation*}
	p(X) = \frac{e^{\beta_0 + \beta_1X_1 + \cdots + \beta_pX_p}}{1+e^{\beta_0 + \beta_1X_1 + \cdots + \beta_pX_p}}
\end{equation*}

\textbf{Equivalent to:}
\begin{equation*}
	\log\left( \frac{p(X)}{1-p(X)} \right)
	= \beta_0 + \beta_1X_1 + \cdots + \beta_pX_p
\end{equation*}
\includegraphics[width = \textwidth, align = c]{Table4-3}
