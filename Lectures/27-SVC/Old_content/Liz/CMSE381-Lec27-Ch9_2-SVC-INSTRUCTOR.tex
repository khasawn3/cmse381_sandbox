\begin{frame}[t]{Separating Hyperplane}

\begin{columns}
\column{.4\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Fig9-2a}
\column{.45\textwidth}
\centering
Require that for every data point:
\begin{align*}
	\beta_0 + \beta_1x_{i1} + \beta_2 x_{i2} + \cdots + \beta_px_{ip} >0 \text{ if } y_i & = 1\\
	\beta_0 + \beta_1x_{i1} + \beta_2 x_{i2} + \cdots + \beta_px_{ip} <0 \text{ if } y_i & = -1
\end{align*}

\vspace{.2in} 

\textit{Equivalently} 

\vspace{.2in} 

Require that for every data point 
\begin{equation*}
	y_i(\beta_0 + \beta_1x_{i1} + \beta_2 x_{i2} + \cdots + \beta_px_{ip}) > 0
\end{equation*}

\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Separating hyperplane becomes a classifier }

	\begin{columns}
	\column{.45\textwidth}
	\centering
	If you have a separating hyperplane:
	\begin{itemize}
		\item Check
		\begin{equation*}
			f(x^*) = \beta_0 + \beta_1x_{1}^* + \beta_2 x_{2}^* + \cdots + \beta_px_{p}^*
		\end{equation*}
		\item If positive, assign $\hat y = 1$
		\item If negative, assign $\hat y = -1$
	\end{itemize}
	\column{.45\textwidth}
	\centering
	\includegraphics[width = \textwidth, align = c]{Fig9-2b}
	\end{columns}
	
\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Maximal margin classifier}

	\begin{columns}
	\column{.4\textwidth}
	\centering
	\includegraphics[width = \textwidth, align = c]{Fig9-3}
	\column{.5\textwidth}
	\centering

	\begin{itemize}
	\item For a hyperplane, the \textit{margin} is the smallest distance from any data point to the hyperplane.
	\item Observations that are closest are called \textit{support vectors}.
	\item The \textit{maximal margin hyperplane} is the hyperplane with the largest margin
	\item The classifier built from this hyperplane is the \textit{maximal margin classifier}.
	\end{itemize}
	
	\end{columns}
	
\end{frame}
%--- Next Frame ---%

\begin{frame}[t]{Mathematical Formulation}

	\begin{columns}
	\column{.65\textwidth}
	\centering
	\includegraphics[width = \textwidth, align = c]{Eq9-9}
	\column{.35\textwidth}
	\centering
	\InstructorList{
		\item Unit normal requirement, we can always write a given hyperplane this way.
	  \item Last eq forces points to be on the correct side of the hyperplane in order for $M>0$
	  \item The product is the distance from the point to the hyperplane 
	\item Making $M$ as big as possible is the maximal margin hyperplane
	}
	
	\end{columns}
	
	\end{frame}

\begin{frame}[t]{Find positive $\e$'s that will satisfy this}

	\vspace{-.3in}
	\begin{equation*}
		\text{Fix } M = \sqrt 2 \qquad y_i(\beta_0 + \beta_1x_{i1} + \beta_2 x_{i2} + \cdots + \beta_px_{ip}) \geq M(1-\e_i)
	\end{equation*}

\begin{columns}
\column{.45\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{SVC-DesmosExample-A_and_C}

\item $A$ is on the correct side of the hyperplane. 
\InstructorList{
\item Left side is already bigger than $M$, so set $\e_i = 0$. 
}
  \item $C$ is on the wrong side of the hyperplane for its label.

  \item $-\frac{\sqrt{2}}{2} = \sqrt{2}(1-\e)$
  \item $\e = 3/2$
}
}

\end{columns}
\end{frame}

\begin{frame}[t]{What is $\e$?}

\begin{columns}
\column{.45\textwidth}
\centering

% \includegraphics[width = \textwidth, align = c]{Eq9-12}
\footnotesize
\InstructorList{
  \item If $\e_i = 0$, then on correct side of margin
	\item If $\e_i>0$ then on the wrong side of margin  (Violated margin)
	\item If $\e_i >1$ then on the wrong side of hyperplane
}

\begin{frame}[t]{What is $C$?}

\begin{columns}
\column{.55\textwidth}
\centering

\includegraphics[width = \textwidth, align = c]{Eq9-12}
\footnotesize
\InstructorList{
\item Bounds sum of $\e_i$, so controls number \& severity of violating margin (budget)
\item $C=0$ means no violations allowed
\item $C>0$ means at most $C$ observations can be on wrong side of hyperplane
\item In previous example, our total of $\e$ was 4, so would be a valid hyperplane for $C$ at most 4. 
}

\centering
