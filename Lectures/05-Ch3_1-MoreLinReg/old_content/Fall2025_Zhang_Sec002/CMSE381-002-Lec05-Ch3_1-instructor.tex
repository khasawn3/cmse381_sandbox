\begin{frame}[c]{Setup}

\begin{columns}
\column{.4\textwidth}
\centering
\begin{itemize}
	\item Predict $Y$ on a single predictor variable $X$
	\begin{equation*}
		Y \approx \beta_0 + \beta_1 X
	\end{equation*}
	\item "$\approx$'' .... ``is approximately modeled as''
\end{itemize}
\column{.4\textwidth}
\begin{itemize}
\item Given $(x_1,y_1),\cdots,(x_n,y_n)$
\item Let $	\hat y_i = \hat \beta_0 + \hat \beta_1 x_i$ be prediction for $Y$ on $i$th value of $X$.
\item $e_i = y_i - \hat y_i$ is the $i$th residual
\end{itemize}
\includegraphics[width = \textwidth]{Fig3-1}
\centering



\end{columns}

\end{frame}

\begin{frame}[c]{Least squares criterion: RSS}

\begin{columns}
\column{.6\textwidth}
\centering
\includegraphics[width = \textwidth]{Fig3-2}
\column{.34\textwidth}
\centering
Residual sum of squares RSS is
\begin{align*}
	RSS &= e_1^2 + \cdots + e_n^2\\
	 	  &= \sum_i (y_i - \hat \beta_0 - \hat \beta_1 x_i)^2
\end{align*}
\end{columns}

\begin{columns}
\column{.5\textwidth}
\centering

\begin{block}{Least squares criterion}
Find $\beta_0$ and $\beta_1$ that minimize the RSS. 
\end{block}

\column{.4\textwidth}
\begin{align*}
	\hat \beta_1 &=
		\frac{\sum_{i=1}^n(x_i-\overline x) (y_i - \overline y)}
			{\sum_{i=1}^n (x_i - \overline x)^2}\\
			\hat \beta_0 & = \overline y - \hat \beta_1 \overline x
\end{align*}
\end{columns}
\end{frame}

\begin{frame}[c]{Bias in estimation}
	{Analogy with mean}

	\InstructorNotes{Everything here is about determining if linear model is doing a good job}
	
	\InstructorNotes{Explain sample vs population.}
\begin{columns}
\column{.4\textwidth}
\centering
\begin{itemize}
	\item Assume a true value $\mu ^*$
	\item An estimate from training data $\hat \mu$
	\item The estimate is unbiased if $E(\hat \mu = \mu^*)$
\end{itemize}

\column{.5\textwidth}
\centering
\begin{itemize}
	\item Sample mean is unbiased for population mean:
	\begin{equation*}
		E(\hat \mu) = E\left(\frac{1}{n}\sum_i X_i \right) = \mu
	\end{equation*}
	\item Standard variance estimate is biased
	\InstructorNotes{(under estimation)}
	\begin{equation*}
		E(\hat \sigma ^2) = E\left[\frac{1}{n} \sum_i(X_i - \overline X)^2\right] \neq \sigma^2
	\end{equation*}
\end{itemize}
\end{columns}

\end{frame}

\begin{frame}[t]{Variance of linear regression estimates }
\begin{columns}
\column{.55\textwidth}
\centering
\centering
\begin{itemize}
	\item Variance of linear regression estimates:
	\begin{align*}
		\SE(\hat \beta_0)^2
				&= \sigma^2 \left[ \frac{1}{n} + \frac{\overline x^2}{\sum_{i=1}^n (x_i - \overline x)^2}\right]\\
		\SE(\hat \beta_1)^2 &= \frac{\sigma^2}{\sum_{i=1}^n (x_i - \overline x)^2}
	\end{align*}
	where $\sigma^2 = \Var(\e)$
	\item Residual standard error is an estimate of $\sigma$
	\begin{equation*}
		RSE = \sqrt{RSS/ (n-2)}
	\end{equation*}
	% Estimate $\sigma$:
	% $\hat \sigma^2 = \frac{RSS}{n-2}$
\end{itemize}
\column{.35\textwidth}
\centering
\InstructorNotes{if data are more spreadout the SE is smaller}

\end{columns}

\begin{frame}[c]{Confidence Interval}

\begin{columns}
\column{.4\textwidth}
% \centering
The 95\% confidence interval for $\beta_1$ approximately takes the form
\begin{equation*}
	\hat \beta_1 \pm 2 \cdot \SE(\hat \beta_1)
\end{equation*}
\column{.4\textwidth}
% \centering
\textbf{Interpretation: }

There is approximately a 95\% chance that the interval
\begin{equation*}
	\left[ 	\hat \beta_1 - 2 \cdot \SE(\hat \beta_1),
	\hat \beta_1 + 2 \cdot \SE(\hat \beta_1)
\right ]
\end{equation*}
will contain $\beta_1$ where we repeatedly approximate $\hat \beta_1$ using repeated samples.
\end{columns}

\InstructorNotes{Same form works for $\beta_0$}
\end{frame}

\begin{frame}[c]{Hypothesis testing}

\begin{columns}

\centering

	\item [ $H_0$:] There is no relationship between $X$ and $Y$
	\InstructorNotes{(null hypothesis)}
	\item [$H_1$:] There is some relationship between $X$ and $Y$
	\InstructorNotes{(alternative hypothesis)}
\end{itemize}
\column{.4\textwidth}
\centering
\InstructorNotes{
\begin{align*}
	H_0: &\beta_1 = 0 \\
	H_A: &\beta_1 \neq 0
\end{align*}
since if $\beta_1 = 0$ then the model reduces to $Y = \beta_0+\e$, and thus $X$ is not associated with $Y$

\begin{frame}[c]{Assessing the accuracy of the module: RSE}
	\centering
\InstructorNotes{Quantify the extent to which the model fits the data}
\begin{columns}
\column{.4\textwidth}
% \centering
\textbf{Residual standard error (RSE):}
\begin{align*}
	RSE &= \sqrt{ \frac{1}{n-2} RSS}\\
	& = \sqrt{ \frac{1}{n-2} \sum_i (y_i - \hat y_i)^2}
\end{align*}

\column{.45\textwidth}
\InstructorList{
\item estimate of the standard deviation of $\e$
\item Avg amount that the response will deviate from the true regression line 
\item  avg amount response will deviate from the true regression line}
% \centering
\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Assessing the accuracy of the module: $R^2$}
	\centering
\begin{columns}
\column{.4\textwidth}
% \centering
\textbf{R squared:}
\begin{align*}
	R^2 &= \frac{TSS-RSS}{TSS} = 1-\frac{RSS}{TSS}
\end{align*}
where total sum of squares is
\begin{equation*}
	TSS = \sum_i (y_i - \overline y)^2
\end{equation*}
\column{.45\textwidth}
\InstructorList{
\item TSS is total variance in the response $Y$, variability before regression
\item RSS amount of variability after the regression
\item $R^2$ is proportion of variability in $Y$ that can be explained using $X$
\item Close to 1, large proportion of varaiability is explained by regression
\item Close to 0, regression does not explain much of the variability in the response
}
\end{columns}

\end{frame}
