\begin{frame}[c]{Least squares criterion: RSS}

\begin{columns}
\column{.6\textwidth}
\centering
\includegraphics[width = \textwidth]{Fig3-2}
\begin{equation*}
	\textrm{sales} \approx \beta_0 + \beta_1 \textrm{TV}
\end{equation*}
\column{.34\textwidth}
\centering
Residual sum of squares RSS is
\begin{align*}
	RSS &= e_1^2 + \cdots + e_n^2\\
	 	  &= \sum_i (y_i - \hat \beta_0 - \hat \beta_1 x_i)^2
\end{align*}
\end{columns}

\begin{columns}
\column{.5\textwidth}
\centering

\begin{block}{Least squares criterion}
Find $\beta_0$ and $\beta_1$ that minimize the RSS. 
\end{block}
\end{columns}
\end{frame}

\begin{frame}[t]{Least squares coefficient estimates}

\begin{columns}
\column{.4\textwidth}
\centering

\InstructorNotes{Minimizing RSS:}
\begin{equation*}
	\min_{\beta_0,\beta_1} \sum_i (y_i - \hat \beta_0 - \hat \beta_1 x_i)^2 
\end{equation*}
\InstructorNotes{There's a typo on their version}

\begin{align*}
\frac{\partial RSS}{\partial \beta_0} & = -2 \sum_i (y_i - \beta_0 - \beta_1x_i) = 0 \\
\frac{\partial RSS}{\partial \beta_1} & = -2 \sum_i x_i(y_i - \beta_0 - \beta_1x_i) = 0
\end{align*}



\column{.4\textwidth}
\centering

\begin{align*}
	\hat \beta_1 &=
		\frac{\sum_{i=1}^n(x_i-\overline x) (y_i - \overline y)}
			{\sum_{i=1}^n (x_i - \overline x)^2}\\
			\hat \beta_0 & = \overline y - \hat \beta_1 \overline x
\end{align*}
