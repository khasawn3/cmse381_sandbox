\begin{frame}[t]{Choices for activation function }

	\begin{columns}
	\column{.45\textwidth}
	\centering
	Sigmoid:
	\begin{equation*}
		g(z) = \frac{e^z}{1+e^z} = \frac{1}{1+e^{-z}}
	\end{equation*}
	% \InstructorList{
	%   \item Converts linear function in probabilities in [0,1]
	% }

	\includegraphics[align = c, width = \textwidth]{Sigmoid-pytorch}
	
	
	\column{.45\textwidth}
	\centering
	ReLU: Rectified linear unit
	\begin{equation*}
		g(z) = (z)_+ =
		\begin{cases}
			0 & \text{if }z<0 \\
			z & \text{else.}
		\end{cases}
	\end{equation*}
	\includegraphics[align = c, width = \textwidth]{ReLU-pytorch}
	\end{columns}
	
	\end{frame}

\begin{frame}[t]{Training the model}
Choose parameters by minimizing RSS, $\sum_{i=1}^n(y_i - f(x_i))^2$ (or other loss function)
\begin{columns}
\column{.45\textwidth}
\centering
\textbf{Chosen in advance:}
\InstructorList{
  \item Number of layers (more on that in a moment)
	\item Number of hidden units
	\item Activation function $g(z)$
}


\column{.45\textwidth}
\centering
\textbf{Tuned by the model:}
\InstructorList{
  \item $\omega$
	\item $\beta$
}

\end{columns}
\InstructorNotes{Taken together, these choices are called the "architecture"}
\end{frame}
