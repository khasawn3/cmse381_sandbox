\begin{frame}[c]{How to get logistic function}

\begin{columns}
\column{.4\textwidth}
\centering
Assume the (natural) log odds (logits) follow a linear model

\begin{equation*}
	\log \left( \frac{p(x)}{1-p(x)} \right) = \beta_0 + \beta_1x
\end{equation*}


\column{.4\textwidth}
\centering
Solve for $p(x)$:
\begin{equation*}
	p(x) = \frac{e^{\beta_0 + \beta_1x}}{1+e^{\beta_0 + \beta_1x}}
\end{equation*}
\end{columns}
\InstructorList{
	% \item The fraction $p/(1-p)$ is called the odds. 
	% \item Can take any value from 0 (high chance of default) to $\infty$ (low chance of default).
	% \item Note that this means the log odds are linear
	\item odds more often associated with bets "doubling down"
	\item human (animals) perceives doubling in physical quantity as 1 unit of perceptual change, log converts former to latter (Weber's Law)
}
\bvfill

Playing with the logistic function: \href{https://www.desmos.com/calculator/cw1pyzzqci}{desmos.com/calculator/cw1pyzzqci}
\end{frame}

\begin{frame}[c]{Multiple Logistic Regression}

\begin{columns}
\column{.65\textwidth}
\centering
\textbf{Multiple features:}
\begin{equation*}
	p(X) = \frac{e^{\beta_0 + \beta_1X_1 + \cdots + \beta_pX_p}}{1+e^{\beta_0 + \beta_1X_1 + \cdots + \beta_pX_p}}
\end{equation*}

\textbf{Equivalent to:}
\begin{equation*}
	\log\left( \frac{p(X)}{1-p(X)} \right)
	= \beta_0 + \beta_1X_1 + \cdots + \beta_pX_p
\end{equation*}

% \column{.25\textwidth}
% \centering

\end{columns}

\begin{frame}[t]{Log odds }
\centering 
Calculated so that log odds between \textit{any pair of} classes is linear.

Specifically, for $Y=k$ vs $Y=K$, we have

\begin{equation*}
	\log\left(
\frac{\Pr(Y = k \mid X = x)}{\Pr(Y = K \mid X = x)}
	\right)
	= \beta_{k0} + \beta_{k1}x_1 + \cdots + \beta_{kp}{x_p}
\end{equation*}
