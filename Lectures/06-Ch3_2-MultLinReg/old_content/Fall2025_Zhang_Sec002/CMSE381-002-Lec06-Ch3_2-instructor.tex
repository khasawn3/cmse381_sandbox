\begin{frame}[c]{Linear Regression with One Variable}

\begin{columns}
\column{.45\textwidth}
\centering
\includegraphics[width = .5\textwidth]{Fig3-1}
\begin{itemize}
	\item Predict $Y$ on a single variable $X$
	\begin{equation*}
		Y \approx \beta_0 + \beta_1 X
	\end{equation*}
	\item Find good guesses for $\hat \beta_0$, $\hat \beta_1$.
\item $	\hat y_i = \hat \beta_0 + \hat \beta_1 x_i$
\item $e_i = y_i - \hat y_i$ is the $i$th residual
\item residual sum of squares RSS $= \sum_i e_i^2$
\end{itemize}
\column{.4\textwidth}
\centering
\includegraphics[width = \textwidth]{Fig3-2}
\begin{itemize}
	\item RSS is minimized at \textit{least squares coefficient estimates}
\begin{align*}
	\hat \beta_1 &=
		\frac{\sum_{i=1}^n(x_i-\overline x) (y_i - \overline y)}
			{\sum_{i=1}^n (x_i - \overline x)^2}\\
			\hat \beta_0 & = \overline y - \hat \beta_1 \overline x
\end{align*}
\end{itemize}

\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[c]{Evaluating the model}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{itemize}
	\item Linear regression is unbiased
	\item Variance of linear regression estimates:
	\begin{align*}
		\SE(\hat \beta_0)
				&= \sigma^2 \left[ \frac{1}{n} + \frac{\overline x^2}{\sum_{i=1}^n (x_i - \overline x)^2}\right]\\
		\SE(\hat \beta_1)^2 &= \frac{\sigma^2}{\sum_{i=1}^n (x_i - \overline x)^2}
	\end{align*}
	where $\sigma^2 = \Var(\e)$
	\item Estimate $\sigma$:
	$\hat \sigma^2 = \frac{RSS}{n-2}$
\end{itemize}

\column{.45\textwidth}
\centering
\begin{itemize}
	\item The 95\% confidence interval for $\beta_1$ approximately takes the form
	\begin{equation*}
		\hat \beta_1 \pm 2 \cdot \SE(\hat \beta_1)
	\end{equation*}
	\item Hypothesis test:
	\begin{itemize}
			\item [ $H_0$:] $\beta_1 = 0$
			\item [$H_a$:] $\beta_1 \neq 0$
			\item Test statistic $	t = \frac{\hat \beta_1 - 0}{\SE(\hat \beta_1)}$
		\end{itemize}
\end{itemize}
\end{columns}

\end{frame}
%--- Next Frame ---%



\begin{frame}[c]{Assessing the accuracy of the model}
	\centering
\begin{columns}
\column{.4\textwidth}
% \centering
\textbf{Residual standard error (RSE):}
\begin{align*}
	RSE &= \sqrt{ \frac{1}{n-2} RSS}\\
\end{align*}
\InstructorNotes{an estimate of $\sigma$. \\also measure lack of fit? ($MSE=RSS/n$) }



\column{.45\textwidth}

\textbf{R squared:}
\begin{align*}
	R^2 &= \frac{TSS-RSS}{TSS} = 1-\frac{RSS}{TSS}
\end{align*}
% where total sum of squares is
\begin{equation*}
	TSS = \sum_i (y_i - \overline y)^2
\end{equation*}
\InstructorList{
\item TSS = total sum of squares
\item Proportion in [0,1]; variability in $Y$ that can be explained using $X$
\item 0 means $Y$ not explained by linear regression on $X$}
\end{columns}

\begin{frame}[c]{Estimation and Prediction}

\begin{columns}
\column{.4\textwidth}
% \centering
Given estimates $\hat \beta_0, \hat \beta_1,\hat \beta_2,\cdots,\hat\beta_p$, prediction is
\begin{equation*}
	\hat y = \hat \beta_0 +  \hat \beta_1x_1 + \cdots + \hat \beta_p x_p
\end{equation*}
\column{.45\textwidth}
% \centering
Minimize the sum of squares
\begin{align*}
	RSS &= \sum_i (y_i - \hat y_i)^2\\
	&= \sum_i (y_i - \hat \beta_0 - \hat \beta_1x_{i1} - \cdots - \hat \beta_p x_{ip})^2
\end{align*}

\begin{frame}[t]{Q1: Hypothesis test}

\begin{columns}
\column{.4\textwidth}
\centering
\begin{itemize}
	\item [$H_0 : $] $\beta_1 = \beta_2 = \cdots = \beta_p = 0$
	\item [$H_a:$] At least one $\beta_j$ is non-zero
\end{itemize}

\column{.4\textwidth}
\centering

\textbf{F-statistic:}
\begin{equation*}
	F = \frac{(TSS-RSS)/p}{RSS/(n-p-1)} \sim F_{p,n- p - 1}
\end{equation*}

\InstructorList{
\item TSS $= \sum_i (y_i - \overline y)^2$
\item RSS $= \sum_i (y_i - \hat y_i)^2$
}

\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[t]{The F-statistic for the hypothesis test}
\begin{columns}
\column{.45\textwidth}
\centering
\begin{equation*}
	F = \frac{(TSS-RSS)/p}{RSS/(n-p-1)} \sim F_{p,n- p - 1}
\end{equation*}

\includegraphics[align = c, width = \textwidth]{640px-F-distribution_pdf}

\column{.45\textwidth}
\centering

\InstructorList{
\item If linear model assumptions are true
\begin{equation*}
	E( RSS / (n- p - 1)) = \sigma^2
\end{equation*}
\item and if $H_0$ is true
\begin{equation*}
	E((TSS-RSS)/p) = \sigma^2
\end{equation*}
\item So if no relationship between response and predictors, $F$ is close to 1
\item If $H_a$ is true,
\begin{equation*}
	E((TSS-RSS)/p) > \sigma^2
\end{equation*}
so then $F > 1$
}
\end{columns}
