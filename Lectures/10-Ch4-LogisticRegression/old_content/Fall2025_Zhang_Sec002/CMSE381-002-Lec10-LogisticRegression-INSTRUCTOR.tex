\begin{frame}[t]{Error rate}



	\begin{columns}
	\column{.5\textwidth}
	\centering
	\begin{itemize}
		\item
	Training data:
	
	$\{ (x_1,y_1),\cdots,(x_n,y_n) \}$ with $y_i$ qualitative
	
	\item Estimate $\hat y = \hat f (x)$
	\item Indicator variable
	% \vspace{.5in}
	
	\InstructorNotes{$\mathrm{I}(y_i \neq \hat y_i)$ is 1 if not the same (misclassified) and 0 else}
	% \item
	\end{itemize}
	
	\column{.4\textwidth}
	Training error rate:
	\begin{equation*}
	\frac{1}{n} \sum_{i = 1}^n \mathrm{I}(y_i \neq \hat y_i)
	\end{equation*}
	
	Test error rate:
	\begin{equation*}
		\mathrm{Ave}(\mathrm{I} (y_0 \neq \hat y_0))
	\end{equation*}
	
	\InstructorNotes{Good classifier: one that minimizes test error rate, GET BETTER EXAMPLE FOR THIS}
	
	% \vspace{1.5in}
	\end{columns}

	\begin{frame}[t]{Bayes error rate}
	
	\begin{columns}
	
	\column{.5\textwidth}
	\centering
	\begin{itemize}
		\item Error at $X = x_0$
		\begin{equation*}
			1 - \max_j \Pr(Y = j \mid X = x_0)
		\end{equation*}
		\item Overall Bayes error:
		\begin{equation*}
			1 - E \left(\max_j \Pr(Y = j \mid X = x_0) \right)
		\end{equation*}
	
		\InstructorNotes{This is the irreduceable error}
	\end{itemize}
	
	\column{.4\textwidth}
	\centering
		\includegraphics[width = \textwidth]{Fig2-13.png}
	\end{columns}
	\InstructorList{
		\item Game: do as good a job as possible estimating the Bayes decision boundary
		\item Classify given observation with highest estimated probability
		}

\begin{frame}[t]{Odds}
	\begin{equation*}
		\frac{p(x)}{1-p(x)}
		= \frac{\Pr(Y = 1 \mid X=x)}{1-\Pr(Y = 1 \mid X=x)}
		= \frac{\Pr(Y = 1 \mid X=x)}{\Pr(Y = 0 \mid X=x)}
	\end{equation*}
\begin{columns}
\column{.4\textwidth}
\centering
\includegraphics[width = .6\textwidth,align = c]{1024px-Probability_vs_odds.png}

\InstructorList{
\item Logistic function is chosen so that log odds are linear
\item Can take any value from 0 (low odds) to $\infty$ (high odds)
}


\column{.4\textwidth}
Examples:
\begin{itemize}
	\item If the probability of default is 90\% what are the odds?
	\begin{itemize}
		\item $p(x) =0.9$ 
		\item $\frac{0.9}{1-0.9} = 9$
	\end{itemize}
	\item   If the odds are 1/3, what is the probability of default?
	\begin{itemize}
	  \item $\frac{p}{1-p} = 1/3$
	  \item $3p = 1-p$
	  \item $4p = 1$
	  \item $p = 1/4$
	\end{itemize}
\end{itemize}
\end{columns}

\begin{frame}[c]{Making the nonlinear linear}

\begin{columns}
\column{.4\textwidth}
\centering
Assume the (natural) log odds (logits) follow a linear model

\begin{equation*}
	\log \left( \frac{p(x)}{1-p(x)} \right) = \beta_0 + \beta_1x
\end{equation*}

Do some algebra and get $p(x)$:
\begin{equation*}
	p(x) = \frac{e^{\beta_0 + \beta_1x}}{1+e^{\beta_0 + \beta_1x}}
\end{equation*}
