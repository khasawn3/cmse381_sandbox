\begin{frame}[c]{Linear Regression with Multiple Variables}

\begin{columns}
\column{.45\textwidth}
\centering
\includegraphics[width = .5\textwidth]{Fig3-4}
\begin{itemize}
	\item Predict $Y$ on a multiple variables $X$
\begin{equation*}
	Y = \beta_0 + \beta_1 X_1 + \beta_2X_2 + \cdots \beta_p x_p + \e
\end{equation*}
	\item Find good guesses for $\hat \beta_0$, $\hat \beta_1,\cdots $.
\item $	\hat y_i = \hat \beta_0 + \hat \beta_1 x_i + \cdots + \hat \beta_px_p$
\end{itemize}
\column{.4\textwidth}
\centering
\begin{itemize}
\item $e_i = y_i - \hat y_i$ is the $i$th residual
\item RSS $= \sum_i e_i^2$
	\item RSS is minimized at \textit{least squares coefficient estimates}
\end{itemize}

\end{columns}

\end{frame}

\section{Extending the linear model}
%=================

\begin{frame}[c]{Assumptions so far }{Back to our Advertising data set }


\begin{equation*}
	\hat Y_{sales}
	= \beta_0
	+ \beta_1 \cdot X_{{TV}}
	+ \beta_2 \cdot X_{{radio}}
	+ \beta_3 \cdot X_{{newspaper}}
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering
Assumed (implicitly) that the effect on sales by increasing one medium is independent of the others.
\column{.45\textwidth}
\centering
What if spending money on radio advertising increases the effectiveness of TV advertising? How do we model it?
\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Interaction Term}
\begin{columns}
\column{.5\textwidth}
\centering
\begin{equation*}
	Y = \beta_0 + \beta_1X_1 + \beta_2 X_2 + \beta_3 X_1X_2 + \e
\end{equation*}

\vspace{.2in}

\begin{align*}
	Y_{sales} &= \beta_0 + \beta_1 X_{TV} + \beta_2 X_{radio}+ \beta_3 X_{radio}X_{TV} + \e\\
	&= \beta_0 + (\beta_1 + \beta_3 X_{radio}) X_{TV} + \beta_2 X_{radio} + \e \\
	&= \beta_0 + \tilde{\beta_1} X_{TV} + \beta_2 X_{radio} + \e
\end{align*}

\column{.35\textwidth}
\centering
\InstructorList{
  \item $\beta_3 X_{radio} X_{TV}$ is the interaction term
  \item $\tilde{\beta_1} = \beta_1 + \beta_3 X_{radio}$, function of $X_2$ association no longer constant between $X_1$ and $Y$}
\end{columns}
\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Interaction term}

\begin{columns}
\column{.45\textwidth}
\centering
\begin{equation*}
	Y = \beta_0 + \beta_1X_1 + \beta_2 X_2 + \beta_3 X_1X_2 + \e
\end{equation*}

\vspace{.2in}

\begin{align*}
	Y_{sales} &= \beta_0 + \beta_1 X_{TV} + \beta_2 X_{radio}+ \beta_3 X_{radio}X_{TV} + \e\\
	&= \beta_0 + (\beta_1 + \beta_3 X_{radio}) X_{TV} + \beta_2 X_{radio} + \e
\end{align*}
\column{.45\textwidth}
\centering
\includegraphics[width = \textwidth, align = c]{Table3-9}
\InstructorList{
\item P-value for interaction term is low, strong evidance for $H_a: \beta_3 \neq 0$.
\item $R^2$ is 96.8\% for this, vs 89.7\% for model without interaction
\item $\beta_3$ the increase in the effectiveness of TV advertising
associated with a one-unit increase in radio advertising 
}
\end{columns}

\end{frame}

\begin{frame}[c]{Hierarchy principle}

\begin{columns}
\column{.45\textwidth}
\centering
Sometimes $p$-value for interaction term is very small, but associated main effects are not.
\column{.45\textwidth}
\centering
The hierarchy principle:
\InstructorList{
\item If we include the interaction term, we should include the main effects no matter what.
\item
}
\end{columns}


\end{frame}
