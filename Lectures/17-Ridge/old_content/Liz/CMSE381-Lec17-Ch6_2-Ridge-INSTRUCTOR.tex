\begin{frame}[t]{Ridge regression}

\begin{columns}
\column{.3\textwidth}
\centering
\textbf{Before:}
\footnotesize
\begin{equation*}
	RSS = \sum_{i=1}^n \left(y_i - \beta_0 - \sum_{j=1}^p \beta_j x_{ij}\right)^2
\end{equation*}

\column{.55\textwidth}
\centering
\textbf{After:}
\begin{equation*}
	\sum_{i=1}^n \left(y_i - \beta_0 - \sum_{j=1}^p \beta_j x_{ij}\right)^2 + \lambda \sum_{j=1}^p \beta_j^2 = RSS + \lambda \sum_{j=1}^p \beta_j^2
\end{equation*}

\end{columns}


\begin{columns}
\column{.45\textwidth}
\centering
\footnotesize
\InstructorList{
\item $\lambda \geq 0$ is a tuning paramter to figure out separately.
\item Second term is called ``shrinkage penalty''
\item $\lambda = 0$ means least squares estimate
}
\column{.45\textwidth}
\centering

\footnotesize
\InstructorList{
\item As $\lambda \to \infty$ impact of shrinkage penalty grows, ridge regression coeff will go towards 0
% \item note different set of estimates every time
\item Not applied to the intercept
\item call coefficients found by ridge regression $\hat \beta_\lambda^R = (\beta_{1,\lambda}^R, \cdots,\beta_{p,\lambda}^R \cdots )$
}
\end{columns}

\end{frame}

\begin{frame}[t]{Scale equivavariance (or lack thereof)}

\begin{columns}
\column{.45\textwidth}
\centering
\textbf{Scale equivariant:} Multiplying a variable by $c$ ($cX_i$) just returns a coefficient multiplied by $1/c$ ($1/c \beta_i$)

\column{.45\textwidth}
\centering
\InstructorList{
\item Ex: \texttt{income variable}.
\item Least squares is scale equivariant
\item Ridge regression very much is not
\item $X_j\hat \beta_{j,\lambda}^R$ depends not only on $\lambda$ but also on values of other predictors
}
\end{columns}

\end{frame}
%--- Next Frame ---%
\begin{frame}[t]{Solution: Standardize predictors}

\begin{equation*}
	\widetilde x_{ij} = \frac{x_{ij}}{
	\sqrt{ \tfrac{1}{n} \sum_{i=1}^n (x_{ij}-\overline x_j)^2}
	}
\end{equation*}
\begin{columns}
\column{.45\textwidth}
\centering
\InstructorList{
\item Denominator is estimated standard deviation of the jth predictor
\item Standardized predictors will all have a standard deviation of one
\item Previous figures showe standardized ridge regression coeffs on the $y$-axis
}
\column{.45\textwidth}
\centering

\end{columns}

\end{frame}
%--- Next Frame ---%

\begin{frame}[c]{Using Cross-Validation to find $\lambda$}
	
	\begin{columns}
	\column{.45\textwidth}
	\centering
	\begin{itemize}
		\item Choose a grid of $\lambda$ values
		\item Compute the ($k$-fold) cross-validation error for each value of $\lambda$
		\item Select the tuning parameter value $\lambda$ for which the CV error
	is smallest.
	  \item The model is re-fit using all of the available observations and the selected value of the tuning parameter.
	\end{itemize}
	\column{.45\textwidth}
	\centering
	
	\end{columns}
	
	\end{frame}

\begin{frame}[t]{Advantages of Ridge }

\begin{columns}
\column{.45\textwidth}
\centering
\textbf{Ridge vs. Least Squares: }
\InstructorList{
\item Previous slide and ability to lower variance
\item Ridge regression works best in situations where the least squares estimates have high variance
\item Can trade off small increase in bias for a large decrease in variance
}
\column{.45\textwidth}
\centering
\textbf{Ridge vs. Subset Selection: }
\InstructorList{
\item So much better computationally!
\item Only fits a single model rather than $2^p$
\item Tricks available to solve all for all values of $\lambda$ simultaneous
}
\end{columns}

\end{frame}
